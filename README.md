This terraform module allow you to create IAM Role for Prisma Tenant to assume.

Variables:

```
variable "aws_account_id" {
  type        = string
  description = "The AWS Account Id where prisma cloud at."
  default     = "111122223333"
}

variable "prisma_cloud_role_name" {
  type        = string
  description = "Provide an role ARN name (Example: PrismaCloudReadOnlyRole or PrismaCloudReadAndWriteRole or PrismaCloudRole)"
  default     = "PrismaCloudRole"
}

variable "external_id" {
  type        = string
  description = "Provide an ExternalID (Example: Abcd123efgh), this value is usually generated by prisma console and it varies per prisma tenant"
  default     = "Abcd123efgh"
}

variable "is_read_only" {
  type        = bool
  description = "If true, use READ ONLY iam policy, otherwise, grant READ and WRITE policy"
  default     = false
}
```
